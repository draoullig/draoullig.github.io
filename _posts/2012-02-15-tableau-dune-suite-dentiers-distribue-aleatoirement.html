La fonction suivante permet de créer un tableau d'entiers distribué aléatoirement.
Les entiers sont issus d'une suite, par exemple 1-50.
Cette fonction est pratique pour les jeux de carte, lotos...

private int[] getTabRandom(final int max) {

final Random random = new Random();
int[] tabelems = new int[max];
int[] tab = new int[max];
for(int i=0; i<max; i++) {

tabelems[i] = i;

}
int variation = max;
for(int j=0; j < max; j++){

int indice = random.nextInt(variation);
tab[j] = tabelems[indice];
tabelems[indice] = tabelems[variation-1];
variation--;

}

return tab;

}
